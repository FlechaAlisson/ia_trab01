function buscaCoresPossiveis(estadoAtual, cores){
	coresPossiveis = []
	for vizinho in estatoAtual.vizinho{
		for cor in cores{
			if(vizinho.cores != cor){
				coresPossiveis.push(cor)
			}
		}
	}
	return coresPossiveis
}
