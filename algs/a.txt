#include <stdio.h>
#include <stdlib.h>
int main(){
    int i, vetor[10];
    int div = 0;
    int n;
	int controle=0;
	for(i=0;i < 10;i++){
		scanf("%d", &vetor[i]);
	}
	for(n=1;n < 10;n++){
		if(vetor[i] % n == 0){
			div++;
		}
		if(div == 2){
			controle=controle+1;
		}
	}
	printf("%d", controle);
	return 0;
}
